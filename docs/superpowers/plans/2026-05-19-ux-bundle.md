# UX Bundle Implementation Plan — Script Preview + Bulk Delete

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship two independent UX wins: **(1)** a live script-preview endpoint + collapsible pane on the Profiles page, **(2)** bulk multi-select + delete on Hotstrings and Hotkeys lists.

**Architecture:**
- Preview: new `GET /api/v1/downloads/{profileId}/preview` endpoint sharing the existing `AhkScriptGenerator` so preview bytes match download bytes. Frontend pane is a `MudExpansionPanel` inside the existing `ChildRowContent` of `Profiles.razor`.
- Bulk delete: `POST /api/v1/{hotstrings|hotkeys}/bulk-delete` with `{ ids }` body and `{ deletedCount, missingIds }` response. Always returns `200 OK` for well-formed authorized requests; partial success is reported in the body, not the status code. `MudDataGrid.MultiSelection` drives the selection state; a conditional toolbar issues the bulk call.

**Tech Stack:** .NET 10, MediatR, Ardalis.Result, FluentValidation, EF Core (SQL Server), Blazor WASM, MudBlazor 9.x, xUnit, FluentAssertions, NSubstitute, Testcontainers (SQL Server), bUnit.

**Spec:** `docs/superpowers/specs/2026-05-19-ux-bundle-design.md`

**Out of scope:** CLI surface area (`AHKFlowApp.CLI`). The spec scopes to API + Blazor only — no CLI commands gain preview or bulk-delete here. `DownloadsApiClient` mirrors live in both UI and CLI projects; only the UI one changes.

## Conventions Recap

- Handlers inject `IAppDbContext`, `ICurrentUser`, `TimeProvider`, generator services directly — no repository pattern.
- Handler returns `Result<T>`; controller maps via `.ToProblemActionResult(this)`.
- Validators registered automatically by `AddValidatorsFromAssembly` in `DependencyInjection.cs` — no manual DI step needed for new validators.
- All endpoints `[Authorize]` + `[RequiredScope("access_as_user")]` (inherited from controller-level attributes).
- Sealed records for DTOs; primary constructors for handlers; file-scoped namespaces.
- Always thread `CancellationToken` through.
- Commit after each task. Conventional commits.

Handler coverage flows through the API integration tests in Task 3 (byte-for-byte match, ownership, 404). No standalone handler-unit-test project exists in `tests/AHKFlowApp.Application.Tests/` for query handlers requiring `IAppDbContext` — the project leans on integration tests for that.

```csharp
// src/Backend/AHKFlowApp.Application/Queries/Downloads/GetProfileScriptPreviewQuery.cs
using AHKFlowApp.Application.Abstractions;
using AHKFlowApp.Application.DTOs;
using AHKFlowApp.Application.Services;
using AHKFlowApp.Domain.Entities;
using Ardalis.Result;
using MediatR;
using Microsoft.EntityFrameworkCore;

namespace AHKFlowApp.Application.Queries.Downloads;

public sealed record GetProfileScriptPreviewQuery(Guid ProfileId) : IRequest<Result<ProfileScriptPreviewDto>>;

internal sealed class GetProfileScriptPreviewQueryHandler(
    IAppDbContext db,
    ICurrentUser currentUser,
    AhkScriptGenerator generator,
    TimeProvider clock)
    : IRequestHandler<GetProfileScriptPreviewQuery, Result<ProfileScriptPreviewDto>>
{
    public async Task<Result<ProfileScriptPreviewDto>> Handle(
        GetProfileScriptPreviewQuery request, CancellationToken ct)
    {
        if (currentUser.Oid is not Guid ownerOid)
            return Result.Unauthorized();

        Profile? profile = await db.Profiles.AsNoTracking().FirstOrDefaultAsync(
            p => p.Id == request.ProfileId && p.OwnerOid == ownerOid, ct);
        if (profile is null)
            return Result.NotFound();

        Guid pid = profile.Id;

        List<Hotstring> hotstrings = await db.Hotstrings.AsNoTracking()
            .Where(h => h.OwnerOid == ownerOid &&
                        (h.AppliesToAllProfiles || h.Profiles.Any(p => p.ProfileId == pid)))
            .ToListAsync(ct);

        List<Hotkey> hotkeys = await db.Hotkeys.AsNoTracking()
            .Where(h => h.OwnerOid == ownerOid &&
                        (h.AppliesToAllProfiles || h.Profiles.Any(p => p.ProfileId == pid)))
            .ToListAsync(ct);

        string content = generator.Generate(profile, hotstrings, hotkeys);

        return Result.Success(new ProfileScriptPreviewDto(
            Script: content,
            HotstringCount: hotstrings.Count,
            HotkeyCount: hotkeys.Count,
            GeneratedAt: clock.GetUtcNow()));
    }
}
```

```csharp
// src/Backend/AHKFlowApp.Application/Commands/Hotstrings/BulkDeleteHotstringsCommand.cs
using AHKFlowApp.Application.Abstractions;
using AHKFlowApp.Application.DTOs;
using AHKFlowApp.Domain.Entities;
using Ardalis.Result;
using FluentValidation;
using MediatR;
using Microsoft.EntityFrameworkCore;

namespace AHKFlowApp.Application.Commands.Hotstrings;

public sealed record BulkDeleteHotstringsCommand(IReadOnlyList<Guid> Ids)
    : IRequest<Result<BulkDeleteResultDto>>;

public sealed class BulkDeleteHotstringsCommandValidator : AbstractValidator<BulkDeleteHotstringsCommand>
{
    public const int MaxIds = 500;

    public BulkDeleteHotstringsCommandValidator()
    {
        RuleFor(x => x.Ids)
            .NotEmpty().WithMessage("At least one id is required.")
            .Must(ids => ids.Count <= MaxIds).WithMessage($"At most {MaxIds} ids per request.");
        RuleForEach(x => x.Ids)
            .NotEqual(Guid.Empty).WithMessage("Empty Guid is not allowed.");
    }
}

internal sealed class BulkDeleteHotstringsCommandHandler(
    IAppDbContext db,
    ICurrentUser currentUser)
    : IRequestHandler<BulkDeleteHotstringsCommand, Result<BulkDeleteResultDto>>
{
    public async Task<Result<BulkDeleteResultDto>> Handle(
        BulkDeleteHotstringsCommand request, CancellationToken ct)
    {
        if (currentUser.Oid is not Guid ownerOid)
            return Result.Unauthorized();

        Guid[] distinctIds = [.. request.Ids.Distinct()];

        List<Hotstring> owned = await db.Hotstrings
            .Where(h => h.OwnerOid == ownerOid && distinctIds.Contains(h.Id))
            .ToListAsync(ct);

        HashSet<Guid> ownedIds = [.. owned.Select(h => h.Id)];
        Guid[] missing = [.. distinctIds.Where(id => !ownedIds.Contains(id))];

        if (owned.Count > 0)
        {
            db.Hotstrings.RemoveRange(owned);
            await db.SaveChangesAsync(ct);
        }

        return Result.Success(new BulkDeleteResultDto(owned.Count, missing));
    }
}
```

```csharp
internal void OnSelectedItemsChanged(HashSet<HotstringEditModel> selected) => _selected = selected;

private async Task BulkDeleteAsync()
{
    Guid[] ids = [.. _selected.Where(item => item.Id is not null).Select(item => item.Id!.Value)];
    if (ids.Length == 0) return;

    bool? confirm = await DialogService.ShowMessageBoxAsync(
        title: "Delete hotstrings?",
        message: $"Delete {ids.Length} hotstring(s)? This cannot be undone.",
        yesText: "Delete", cancelText: "Cancel");
    if (confirm != true) return;

    ApiResult<BulkDeleteResultDto> result = await Api.BulkDeleteAsync(ids, _cts.Token);
    if (result.IsSuccess)
    {
        int deleted = result.Value!.DeletedCount;
        int missing = result.Value.MissingIds.Count;
        string msg = missing > 0
            ? $"Deleted {deleted} hotstring(s); {missing} were already gone."
            : $"Deleted {deleted} hotstring(s).";
        Snackbar.Add(msg, Severity.Success);
        _selected = [];
        if (_grid is not null) await _grid.ReloadServerData();
    }
    else
    {
        Snackbar.Add(ApiErrorMessageFactory.Build(result.Status, result.Problem), Severity.Error);
    }
}
```
